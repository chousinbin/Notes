# 归并排序



## 归并排序

①确定分界点 mid=q[(l+r)/2]

②递归排序 left right

③归并 合二为一

```c++
#include<bits/stdc++.h>
using namespace std;
const int N=1e6+10;
int n,q[N],ans[N];
void merge_sort(int q[],int l,int r)
{
    if(l>=r) return;
    int mid=l+r>>1;
    merge_sort(q,l,mid),merge_sort(q,mid+1,r);
    int i=l,j=mid+1,k=0;
    while(i<=mid && j<=r)
    {
        if(q[i]<q[j]) ans[k++]=q[i++];
        else ans[k++]=q[j++];
    }
    while(i<=mid) ans[k++]=q[i++];
    while(j<=r) ans[k++]=q[j++];
    for(i=l,j=0;i<=r;i++,j++) q[i]=ans[j];
}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++) cin>>q[i];
    merge_sort(q,0,n-1);
    for(int i=0;i<n;i++) cout<<q[i]<<' ';
    return 0;
}
```

## 归并求逆序对

```c++
#include<bits/stdc++.h>
using namespace std;
const int N=1e6+10;
int q[N],ans[N];
long long merge_sort(int l,int r)
{
    if(l>=r) return 0;
    int mid=l+r>>1;
    //递归相加
    long long res=merge_sort(l,mid)+merge_sort(mid+1,r);
    int i=l,j=mid+1,k=0;
    while(i<=mid && j<=r)
    {
        if(q[i]<=q[j]) ans[k++]=q[i++];
        else
        {
            ans[k++]=q[j++];
            res+=mid-i+1;
            //i和i后边的都能与j构成逆序对
            //mid-i+1是左边能构成逆序对数量
            //递归可以分成很小的片段
            //把同是左边或右边的包含了进去
        }
    }
    while(i<=mid) ans[k++]=q[i++];
    while(j<=r) ans[k++]=q[j++];
    for(i=l,j=0;i<=r;i++,j++) q[i]=ans[j];
    return res;
}
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++) cin>>q[i];
    cout<<merge_sort(0,n-1);
    return 0;
}
```
